                        start: 0,

                    },
                },
                // you can have multiple media queries, the last one matched will be applied
            },
        },
    },

<h2>8. Custom transition</h2>

<div 
    data-supersonic-driver="example-8-driver"
    data-supersonic-type="start">
    I'm a driver, I animate "rotate" with custom transition.
</div>

<div class="example-8-element">
    I'm an animatable element.
</div>
<div
    data-supersonic-driver="example-8-driver"
    data-supersonic-type="end">
    I'm a driver, I finish an animation at appearance.
</div>
new TheSupersonicPluginForScrollBasedAnimation({
    drivers: {
        "example-8-driver": {
            hooks: {
                onInit(driver) {
                    driver.data.easedProgress = 0 // you can use "data" to allow hooks to communicate
                },
                onBeforeRender(driver) {
                    let delta = (driver.progress - driver.data.easedProgress) * 0.1

                    if (Math.abs(delta) > 0.0001) driver.progress = driver.data.easedProgress += delta
                },
                // you can apply any easing from easings.net using "driver.progress"
                // if you have simple transition, you can just use "transition" CSS property on target element without js calculations
            },
            properties: {
                rotate: {
                    start: 0,
                    end: 360,
                    unit: "deg",
                    elements: [".example-8-element"],
                },
            },
        },
    },
})

    I'm a driver, I animate "translateY" on 3 elements.
</div>

<div class="example-9-element-1">
    I'm animating without easing.
</div>
<div class="example-9-element-2">
    I'm animating with easeInOutQuart.
</div>
<div class="example-9-element-3">
    I'm animating with easeInOutQuad.
</div>

<div
    data-supersonic-driver="example-9-driver"
    data-supersonic-type="end">
    I'm a driver, I finish an animation at appearance.
</div>
new TheSupersonicPluginForScrollBasedAnimation({
    drivers: {
        "example-9-driver": {
            properties: {
                translateY: {
                    start: 0,
                    end: 300,
                    unit: "px",
                    elements: [".example-9-element-1", ".example-9-element-2", ".example-9-element-3"],
                },
            },
        },
    },
    elements: {
        ".example-9-element-2": {
            hooks: {
                onAddProperty(element, property) {
                    const _property = { ...property } // we need to create a copy of "property" because there is actually only one instance of "property"
                    const x = _property.driver.progress
                    const result = x < 0.5 ? 8 * x * x * x * x : 1 - Math.pow(-2 * x + 2, 4) / 2

                    _property.value *= result
                    element.properties.set(_property.cssProperty, _property) // we replace the original "property" in an element and the rest is magic
                },
            },
        },
        ".example-9-element-3": {
            hooks: {
                onAddProperty(element, property) {
                    const _property = { ...property }
                    const x = _property.driver.progress
                    const result = x < 0.5 ? 2 * x * x : 1 - Math.pow(-2 * x + 2, 2) / 2

                    _property.value *= result

                    element.properties.set(_property.cssProperty, _property)
                },
            },
        },
    },
})

import { Scene, WebGLRenderer, AnimationMixer, Color, AmbientLight } from "three"

new TheSupersonicPluginForScrollBasedAnimation({
    drivers: {
        "example-10-driver": {
            hooks: {
                onInit(driver) {
                    const container = document.querySelector(".example-10-element")
                    const scene = new Scene()
                    const renderer = new WebGLRenderer({ antialias: true, canvas: container.querySelector("canvas") })

                    scene.background = new Color(0x79bde7)
                    scene.add(new AmbientLight(0xc0d5ff, 0.3))

                    renderer.setPixelRatio(window.devicePixelRatio)
                    renderer.setSize(container.clientWidth, container.clientHeight)

                    const loader = new GLTFLoader()
                    loader.loadAsync(modelUrl).then((gltf) => {
                        const camera = gltf.cameras[0]
                        camera.aspect = container.clientWidth / container.clientHeight
                        camera.fov = 70
                        camera.updateProjectionMatrix()

                        const mixer = new AnimationMixer(gltf.scene)
                        const action = mixer.clipAction(gltf.animations[0])
                        action.play()

                        driver.data = { container, scene, renderer, easedProgress, camera, mixer }

                        scene.add(gltf.scene)
                    })
                },
                onAfterRender(driver) {
                    if (driver.data.camera) {
                        const { renderer, scene, camera } = driver.data
                        driver.data.mixer.setTime(1.99 * driver.progress)
                        renderer.render(scene, camera)
                    }
                },
                onUpdateLimits(driver) {
                    if (driver.data.camera) {
                        const { container, camera, renderer, scene } = driver.data

                        camera.aspect = container.clientWidth / container.clientHeight
                        camera.updateProjectionMatrix()

                        renderer.setSize(container.clientWidth, container.clientHeight)

                        renderer.render(scene, camera)
                    }
                },
            },
        },
    },
})

<h2>11. Custom start and end</h2>

<div 
    data-supersonic-driver="example-11-driver" 
    data-supersonic-type="start">
    I'm a driver, I animate "translateY" with custom start and end.
</div>

<div class="example-11-element">
    I'm an animatable element.<br />
    I have width: 50%, so my height is dynamic.<br />
    My initial "translateY" is "-height" and i animate to "window.innerHeight - height".<br />
    Resize window to see how cool my master is.
</div>
<div
    data-supersonic-driver="example-11-driver"
    data-supersonic-type="end">
    I'm a driver, I finish an animation at appearance.
</div>
new TheSupersonicPluginForScrollBasedAnimation({
    drivers: {
        "example-11-driver": {
            properties: {
                translateY: {
                    hooks: {
                        onInit(property) {
                            // start() is called each frame every 16ms so you want to cache everything
                            const element = document.querySelector(".example-7-element")
                            property.data = {
                                element,
                                elementHeight: element.clientHeight,
                                windowHeight: window.innerHeight,
                            }
                        },
                        onUpdateLimits(property) {
                            // when window resizes, update cache
                            property.data.elementHeight = property.data.element.clientHeight
                            property.data.windowHeight = window.innerHeight
                        },
                    },
                    start(property) {
                        // use cached values, not directly "element.clientHeight" because it will lead to bad performance
                        return -property.data.elementHeight
                    },
                    end(property) {
                        return property.data.windowHeight - property.data.elementHeight
                    },
                    unit: "px",
                    elements: [".example-7-element"],
                },
            },
        },
    },
})
